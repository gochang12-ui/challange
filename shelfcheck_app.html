const FOLDER_ID = '11czaapsdywz24XhZJsrG0aimmrviFkiY'; // 폴더명 찾거나 없으면 생성 (중복 방지) function getOrCreateFolder(parent, name) { var it = parent.getFoldersByName(name); return it.hasNext() ? it.next() : parent.createFolder(name); } // shelfType → 한글 폴더명 var TYPE_LABEL = {main:'본매대', cold:'냉장매대', event:'행사매대'}; function doPost(e) { try { var d = JSON.parse(e.postData.contents); // syncUp 요청은 별도 처리 if (d.action === 'syncUp') { return handleSyncUp(d); } // 사진 업로드 처리 var root = DriveApp.getFolderById(FOLDER_ID); // 1단계: 점포명_날짜 폴더 var storeFolder = getOrCreateFolder(root, d.storeName + '_' + d.date); // 2단계: 매대 종류 하위폴더 (본매대 / 냉장매대 / 행사매대) var typeLabel = TYPE_LABEL[d.shelfType] || d.shelfType || '기타'; var typeFolder = getOrCreateFolder(storeFolder, typeLabel); var b64 = d.imageData.split(',')[1]; var blob = Utilities.newBlob(Utilities.base64Decode(b64), d.mimeType, d.fileName); var file = typeFolder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return ContentService .createTextOutput(JSON.stringify({success:true, fileId:file.getId(), url:file.getUrl()})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({success:false, error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
